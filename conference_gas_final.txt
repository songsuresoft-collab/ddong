/**
 * [컨퍼런스 동기화용 구글 앱스 스크립트 - 배포용 최종본]
 */

function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName("Conference_List");
    if (!sheet) return ContentService.createTextOutput("[]").setMimeType(ContentService.MimeType.JSON);

    const data = sheet.getDataRange().getValues();
    if (data.length <= 1) return ContentService.createTextOutput("[]").setMimeType(ContentService.MimeType.JSON);

    const headers = data[0];
    const items = data.slice(1).map(row => {
      let obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });
    return ContentService.createTextOutput(JSON.stringify(items)).setMimeType(ContentService.MimeType.JSON);
  } catch (f) {
    return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    if (body.action === 'sync') {
      return ContentService.createTextOutput(updateConferenceSheet(body.data)).setMimeType(ContentService.MimeType.TEXT);
    }
    return ContentService.createTextOutput("Unknown Action").setMimeType(ContentService.MimeType.TEXT);
  } catch (err) {
    return ContentService.createTextOutput("Error: " + err.toString()).setMimeType(ContentService.MimeType.TEXT);
  }
}

function updateConferenceSheet(eventData) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName("Conference_List") || ss.insertSheet("Conference_List");
  
  sheet.clear(); 
  const headers = [["region", "name", "keywords", "date", "location", "search_date"]];
  sheet.getRange(1, 1, 1, 6).setValues(headers).setBackground("#f3f3f3").setFontWeight("bold");

  // 데이터 정규화: 배열이 아닐 경우 내부 리스트 찾기
  let list = [];
  if (Array.isArray(eventData)) {
    list = eventData;
  } else if (eventData && typeof eventData === 'object') {
    // 객체 내부에 배열이 있는지 확인 (예: { conferences: [...] })
    for (let key in eventData) {
      if (Array.isArray(eventData[key])) {
        list = eventData[key];
        break;
      }
    }
    // 만약 배열이 없고 객체 하나만 온 경우
    if (list.length === 0 && eventData.name) {
      list = [eventData];
    }
  }

  if (list.length === 0) return "Success: No valid array found to sync";

  const rows = list.map(ev => [
    ev.region || "Korea",
    ev.name || "Untitled",
    ev.keywords || "",
    ev.date || "TBD",
    ev.location || ev.place || "N/A",
    new Date()
  ]);

  sheet.getRange(2, 1, rows.length, 6).setValues(rows);
  sheet.autoResizeColumns(1, 6);

  return "Success: Synced " + rows.length + " conferences";
}
